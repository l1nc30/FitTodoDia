package com.dlynce.fittododia.ui.nav

sealed class Route(val path: String) {
    data object Home : Route("home")
    data object Agenda : Route("agenda")
    data object Treino : Route("treino")
    data object Progresso : Route("progresso")
    data object Perfil : Route("perfil")

    data object EditWorkout : Route("edit_workout/{dayId}") {
        fun create(dayId: Int) = "edit_workout/$dayId"
    }

    data object ExerciseLibrary : Route("exercise_library/{dayId}") {
        fun create(dayId: Int) = "exercise_library/$dayId"
    }

    data object AddExercise : Route("add_exercise/{dayId}/{exerciseId}") {
        fun create(dayId: Int, exerciseId: Long) = "add_exercise/$dayId/$exerciseId"
    }

    // ✅ NOVO: editar item já adicionado (workoutExerciseId)
    data object EditWorkoutItem : Route("edit_workout_item/{dayId}/{rowId}") {
        fun create(dayId: Int, rowId: Long) = "edit_workout_item/$dayId/$rowId"
    }
}
